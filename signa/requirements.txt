# Self-contained single-host setup.
#
# WebODM installs this into the plugin's per-plugin site-packages
# (<MEDIA_ROOT>/plugins/signa/site-packages) when the plugin is enabled. That
# path lives on the media volume, which the webapp AND the worker share, so the
# Celery worker — where detection actually runs — can import cv2 from it
# (gcp_detect.py adds the path to sys.path as a fallback). numpy already ships
# with WebODM and is reused, so this only pulls OpenCV.
#
# This does NOT cover distributed setups where the worker runs on another host
# without the shared volume — use the docker/ worker image there (the detection
# error message points to it).
opencv-contrib-python-headless==4.10.0.84

# Shared, GUI-free ArUco detection core (single source of truth for the
# dictionaries and the detector, shared with Mensura). Lives in this repo under
# ../signa-core. Until it is published to PyPI, install it into the plugin's
# site-packages alongside this file (or build the worker image with it); the
# plugin imports `signa_core` at load time.
signa-core>=0.1.0
